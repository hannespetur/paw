 <- pf =                    (-1, 0.2, 1.4, 2.6, 3.8, 5, 6.2, 7.4)
 <- qf =                    (0, 0.5, 1, 1.5, 2, 2.5, 3, 3.5)
 <- rf =                    (2, 2.7, 3.4, 4.1, 4.8, 5.5, 6.2, 6.9)
 -> bs::clamp(pf, qf, rf) = (0, 0.5, 1.4, 2.6, 3.8, 5, 6.2, 6.9)

 <- xf =                    3
 <- yf =                    -3
 <- zf =                    2
 -> bs::clamp(xf, yf, zf) = 2
